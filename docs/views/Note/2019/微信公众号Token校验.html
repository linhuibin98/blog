# 公众号的一些数据

  • AppID：应用的ID，不能改

  • AppSecret：应用的密码

  • Token：校验对方服务器身份（自己设置,需要和后台token一致）

  • EncodingAESKey密钥：公众号消息解码密钥

# 校验Token

配置后台服务器时,会发送get请求到你的服务器校验token,确认请求是否是微信端发送过来的

  • 开发者通过检验signature对请求进行校验

  • 若确认此次GET请求来自微信服务器，请原样返回echostr参数内容，则接入生效，成为开发者成功，否则接入失败。

# 加密/校验流程

  1. 将token、timestamp、nonce三个参数进行字典序排序(sort())

  2. 将三个参数字符串拼接成一个字符串进行sha1加密(哈希Hash)

  3. 开发者获得加密后的字符串可与signature对比，标识该请求来源于微信

// 基于koa2搭建服务,koa-router路由管理

const Koa = require('koa2');
const Router = require('koa-router');
const bodyParser = require('koa-bodyparser');
const config = require('./config');
const crypto = require('crypto');

const app = new Koa();
app.use(bodyParser());
const router = new Router();

// 基于crypto封装sha1加密
function sha1(str) {
  return crypto.createHash('sha1').update(str, 'utf-8').digest('hex');
}

function checkSignature(opts) {
  let {signature, timestamp, nonce} = opts;
  // 添加到数组中
  let arr = [config.token, timestamp, nonce];
  // 排序
  arr.sort();  
  // 拼接成一个字符串
  let temStr = arr.join('');
  // 加密
  temStr = sha1(temStr);
  console.log(signature,'/r/n',temStr)
  if (signature === temStr) {
    return true;
  } else {
    return false;
  }
}

// 微信端发送get请求,校验token
router.get('/weixin', async (ctx, next) => {
  let query = ctx.request.query;
  if(checkSignature(query)) {
    ctx.body = query.echostr;
  } else {
    ctx.body = false;
  }
  await next();
})

app.use(router.routes());

app.listen(3000, () => {
  console.log('服务已经启动');
})
